#name druid_death_brave_6

; 5045 - Spirit of Solomon, 5 will, buff
; 5201 - Trauma, 3 might, add 1 mcharge
; 5204 - Wither, 4 will, add wcharge
; 5303 - Malice Blast, 1 will, need 1-3 wcharge
; 32760 - ranged melee
; 32766 - melee
; 5234 - Deadly Shot, 1 might, need 1-3 mcharge  (not found in ability scan)

set var 0
use 5045

:begin
call waitfortarget
use 32760

getwillcharge var
if var >= 2
  use 5303
  wait 1000
endif

getwill var
if var >= 4
  use 5204
  wait 2000
  goto begin
endif

getmightcharge var
if var >= 2
  use 5234
  wait 1000
endif

getmight var
if var >= 3
  use 5201
  wait 2000
endif

goto begin

:waitfortarget
has_target var
if var = 0
  wait 2000
  goto waitfortarget
endif
return
